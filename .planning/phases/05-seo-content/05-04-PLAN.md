**src/components/seo/JsonLd.tsx** — Create the JSON-LD injector component:

```typescript
interface JsonLdProps {
  data: Record<string, unknown> | Record<string, unknown>[]
}

export default function JsonLd({ data }: JsonLdProps) {
  // Sanitize to prevent XSS: replace < with Unicode escape
  const json = JSON.stringify(data).replace(/</g, '\\u003c')
  return (
    <script
      type="application/ld+json"
      dangerouslySetInnerHTML={{ __html: json }}
    />
  )
}
```

Create the directory `src/components/seo/` if it doesn't exist.

CRITICAL: Always replace `<` with `\u003c` in the serialized JSON to prevent XSS via script injection.

**src/lib/seo.ts** — Create schema builder functions (approximately 150-200 lines):

Import `company` from config, `seo` from config, `testimonials` from config.

1. **buildLocalBusinessSchema(options?)** — Returns LocalBusiness schema:
   - `@context`: 'https://schema.org'
   - `@type`: 'LocalBusiness'
   - `name`, `description`, `url`, `telephone`, `email` from company config
   - `logo`: absolute URL from company.url + seo.ogImage
   - `image`: same as logo
   - `address`: PostalAddress from company.address
   - `openingHoursSpecification`: Mon-Fri 8am-6pm, Sat 9am-2pm (from company.hours)
   - `sameAs`: array of social URLs (filter out '#' placeholders)
   - `aggregateRating`: compute from testimonials.all — ratingValue = average of all ratings, reviewCount = count
   - Optional `areaServed`: if `options?.cityName` provided, add `{ '@type': 'City', 'name': options.cityName }`

2. **buildServiceSchema(service)** — Returns Service schema:
   - `@context`: 'https://schema.org'
   - `@type`: 'Service'
   - `name`: service.name
   - `description`: service.meta.description
   - `provider`: `{ '@type': 'LocalBusiness', 'name': company.name }`
   - `areaServed`: `{ '@type': 'State', 'name': 'TX' }` (from company.address.state)
   - `url`: absolute URL company.url + '/' + service.slug

3. **buildFAQSchema(items)** — Returns FAQPage schema:
   - `@context`: 'https://schema.org'
   - `@type`: 'FAQPage'
   - `mainEntity`: map items to `{ '@type': 'Question', 'name': q, 'acceptedAnswer': { '@type': 'Answer', 'text': a } }`

4. **buildAggregateRatingSchema()** — Returns AggregateRating (for embedding in LocalBusiness):
   - Compute from `testimonials.all`: average rating, review count
   - Return `{ '@type': 'AggregateRating', ratingValue: avg.toFixed(1), reviewCount: count }`
   - Return null if no testimonials

5. **buildBreadcrumbSchema(crumbs: { name: string; url: string }[])** — Returns BreadcrumbList:
   - `@context`: 'https://schema.org'
   - `@type`: 'BreadcrumbList'
   - `itemListElement`: map crumbs to ListItem with position (1-indexed), name, and item (absolute URL: company.url + crumb.url)

6. **buildBlogPostingSchema(post)** — Returns BlogPosting:
   - `@context`: 'https://schema.org'
   - `@type`: 'BlogPosting'
   - `headline`: post.title
   - `description`: post.excerpt
   - `datePublished`: post.date
   - `author`: `{ '@type': 'Organization', 'name': company.name }`
   - `publisher`: `{ '@type': 'Organization', 'name': company.name, 'logo': { '@type': 'ImageObject', 'url': absolute logo URL } }`
   - `image`: absolute URL for post cover image
   - `url`: company.url + '/resources/' + post.slug
   - `keywords`: post.tags

Export all functions as named exports.

NOTE: If `ResourcesPost.tsx` already has an inline BlogPosting JSON-LD implementation (from Phase 4), this replaces it with the shared seo.ts function. Check and refactor in Task 2.

3. Create a function to discover all dynamic routes (CANNOT use import.meta.glob — this is Node context):
   ```typescript
   function getRouteList(): string[] {
     // Static pages
     const staticRoutes = [
       '/', '/services', '/projects', '/testimonials',
       '/about', '/contact', '/service-areas', '/ava',
       '/financing', '/resources',
     ]

     // Service routes — read services config
     // NOTE: Cannot import ESM modules directly in vite.config.ts
     // Use a hardcoded list matching src/config/services.ts slugs
     // OR use a synchronous approach to read the config
     const serviceRoutes = ['/roofing', '/siding', '/storm-damage']

     // City routes — read from service-areas.ts (parse file or hardcode)
     // Best approach: read the TypeScript source file and extract slugs
     // Fallback: maintain a list that matches the config
     let cityRoutes: string[] = []
     try {
       const saFile = readFileSync(resolve('src/config/service-areas.ts'), 'utf-8')
       const slugMatches = saFile.matchAll(/slug:\s*'([^']+)'/g)
       cityRoutes = [...slugMatches].map(m => `/service-areas/${m[1]}`)
     } catch { /* fallback: empty */ }

     // Blog routes — read markdown files
     let blogRoutes: string[] = []
     try {
       const blogDir = resolve('src/content/blog')
       blogRoutes = readdirSync(blogDir)
         .filter(f => f.endsWith('.md'))
         .map(f => {
           const { data } = matter(readFileSync(resolve(blogDir, f), 'utf-8'))
           return data.published !== false ? `/resources/${data.slug}` : null
         })
         .filter(Boolean) as string[]
     } catch { /* no blog posts */ }

     // Portfolio routes — read projects.ts
     let portfolioRoutes: string[] = []
     try {
       const projFile = readFileSync(resolve('src/config/projects.ts'), 'utf-8')
       const slugMatches = projFile.matchAll(/slug:\s*'([^']+)'/g)
       portfolioRoutes = [...slugMatches].map(m => `/portfolio/${m[1]}`)
     } catch { /* fallback: empty */ }

     return [...staticRoutes, ...serviceRoutes, ...cityRoutes, ...blogRoutes, ...portfolioRoutes]
   }
   ```

6. **If vite-plugin-sitemap fails** (generates empty sitemap or errors with Vite 7), implement a custom Vite plugin following the existing `copyAndOptimizeImages` pattern in vite.config.ts:
   ```typescript
   function generateSitemap(): Plugin {
     return {
       name: 'generate-sitemap',
       enforce: 'post',
       apply: 'build',
       async closeBundle() {
         const fs = await import('fs/promises')
         const path = await import('path')
         const hostname = 'https://example.com'
         const routes = getRouteList()
         const excludeSet = new Set(['/thank-you', '/404', '/ava'])
         const filteredRoutes = routes.filter(r => !excludeSet.has(r))

         // Generate sitemap XML
         const urlEntries = filteredRoutes.map(route =>
           `  <url>\n    <loc>${hostname}${route}</loc>\n    <changefreq>weekly</changefreq>\n    <priority>0.8</priority>\n  </url>`
         ).join('\n')

         const sitemap = `<?xml version="1.0" encoding="UTF-8"?>\n<urlset xmlns="http://www.sitemaps.org/schemas/sitemap/0.9">\n${urlEntries}\n</urlset>`

         const robots = `User-agent: *\nAllow: /\n\nSitemap: ${hostname}/sitemap.xml`

         const distDir = path.resolve('dist')
         await fs.writeFile(path.join(distDir, 'sitemap.xml'), sitemap, 'utf-8')
         await fs.writeFile(path.join(distDir, 'robots.txt'), robots, 'utf-8')
         console.log(`[generate-sitemap] Generated sitemap.xml with ${filteredRoutes.length} URLs`)
         console.log(`[generate-sitemap] Generated robots.txt`)
       }
     }
   }
   ```
   Add `generateSitemap()` to the plugins array and remove `Sitemap(...)` if it failed.
